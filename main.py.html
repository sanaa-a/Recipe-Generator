import requests
from pprint import pprint

# extract from Edamam
ingredient = input("What ingredient would you like to use?")
app_id = '973abd68'
app_key = '3b20a6a478e9ada14d4b18f9e95c6daa'
url = 'https://api.edamam.com/search?q={}&app_id={}&app_key={}'.format(ingredient, app_id, app_key)
response = requests.get(url)
print(response)
recipes = response.json()
print(recipes)
output = recipes['hits']

name_of_recipes = []
calories = []
cuisine_types = []
dish_types = []
tags = []
ingredients = []
instructions = []

for option in output:
    name_of_recipes.append(option['recipe']['label'])
    calories.append(option['recipe']['totalNutrients']['ENERC_KCAL']['quantity'])
    cuisine_types.append(option['recipe']['cuisineType'][0])
    dish_types.append(option['recipe']['dishType'][0])
    tags.append(option['recipe']['healthLabels'])
    ingredients.append(option['recipe']['ingredientLines'])
    instructions.append(option['recipe']['url'])
compiled = list(zip(name_of_recipes, calories, cuisine_types, dish_types, tags, ingredients, instructions))
